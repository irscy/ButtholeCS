            if (!weaponHolderSprite.FlipH)
            {
                definedWeapon.Position = new Vector2(-14.972f, -38);
                weaponSprite.Scale = new Vector2(0.3f, 0.3f);
                RotationDegrees = 11;
            }
            //if flipped

                definedWeapon.Position = new Vector2(37.676f, -38);
                weaponSprite.Scale = new Vector2(-0.3f, 0.3f);
                definedWeapon.RotationDegrees = -11;
            }
